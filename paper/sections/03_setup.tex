\subsection{The Effective Gradient Ratio}
\begin{definition}[EGR]
\label{def:egr}
At training step $t$ with parameters $(\thetap_t, \phip_t)$, the Effective Gradient Ratio is
\[
    \EGR(t) = \frac{\norm{\nabla_\thetap \loss(\thetap_t, \phip_t)}}{\norm{\nabla_\phip \loss(\thetap_t, \phip_t)}}.
\]
\end{definition}
